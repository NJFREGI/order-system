        .header-info { background: #fff; padding: 15px; margin-bottom: 10px; border-bottom: 1px solid #ebedf0; }

<body>
<div id="app">
    <div class="header-info">
        <van-row justify="space-between" align="center">
            <van-col span="16">
                <h2 style="margin:0; font-size: 18px;">{{ shopName }}</h2>
                <span class="deadline-tag">毎日 {{ deadlineTime }} 締切</span>
            </van-col>
            <van-col span="8" style="text-align: right;">
                <van-button size="small" icon="clock-o" @click="showHistory = true">注文履歴</van-button>
            </van-col>
        </van-row>
        
        <van-notice-bar v-if="!lastOrderConfirmed" left-icon="warning-o" color="#ed6a0c" background="#fffbe8" wrapable scrollable="false">
            前回分の受取確認が未完了です。確認後に次回の注文が可能になります。
            <template #right-icon>
                <van-button size="mini" type="warning" @click="showConfirmPanel = true">確認する</van-button>
            </template>
        </van-notice-bar>
    </div>

    <div style="padding: 0 10px 100px 10px;">
        <van-card
            v-for="item in products"
            :key="item.id"
            :price="item.price"
            :desc="item.spec"
            :title="item.name"
            :thumb="item.image"
            :class="['product-card', item.stock <= 0 ? 'out-of-stock' : '']"
        >
            <template #tags>
                <van-tag plain type="primary">税率 {{ item.taxRate }}%</van-tag>
                <van-tag v-if="item.stock <= 0" type="danger" style="margin-left: 5px;">欠品中</van-tag>
            </template>
            <template #footer>
                <van-button v-if="item.stock > 0" size="small" type="primary" icon="plus" round @click="openSpec(item)">追加</van-button>
                <van-button v-else size="small" disabled round>入荷待ち</van-button>
            </template>
        </van-card>
    </div>

    <van-action-sheet v-model:show="showSpecPanel" title="注文内容">
        <div style="padding: 20px;">
            <h3>{{ activeItem.name }} ({{ activeItem.spec }})</h3>
            <p>単価: ￥{{ activeItem.price }} (税込 {{ calcTaxedPrice(activeItem) }})</p>
            <van-divider />
            <div style="display:flex; justify-content: space-between; align-items: center;">
                <span>数量</span>
                <van-stepper v-model="tempQty" :max="activeItem.stock" integer />
            </div>
            <van-field v-model="tempNote" label="備考" placeholder="特別な要望があれば記入" style="margin-top:15px; border:1px solid #eee" />
            <van-button type="primary" block round style="margin-top:20px;" @click="addToCart">カートに入れる</van-button>
        </div>
    </van-action-sheet>

    <van-submit-bar 
        :price="cartTotalPrice * 100" 
        button-text="注文を確定する" 
        @submit="onSubmit" 
        :disabled="!lastOrderConfirmed || cart.length === 0"
        currency="￥"
    >
        <template #top>
            <van-notice-bar v-if="cart.length > 0" scrollable text="※ 締切時間までは何度でも追加?変更可能です。"></van-notice-bar>
        </template>
    </van-submit-bar>

    <van-popup v-model:show="showConfirmPanel" position="bottom" style="height: 60%;">
        <van-nav-bar title="前回商品の受取確認" />
        <div style="padding:15px;">
            <p style="font-size:14px; color:#666;">届いた商品にチェックを入れてください：</p>
            <van-checkbox-group v-model="confirmedItems">
                <van-cell-group inset>
                    <van-cell v-for="item in lastOrderItems" :key="item.id" :title="item.name + ' x ' + item.qty" clickable @click="toggleCheck(item.id)">
                        <template #right-icon>
                            <van-checkbox :name="item.id" />
                        </template>
                    </van-cell>
                </van-cell-group>
            </van-checkbox-group>
            <van-button type="success" block round style="margin-top:20px;" @click="confirmReceipt">受取を完了する</van-button>
        </div>
    </van-popup>
</div>

<script>
    const { createApp, ref, computed } = Vue;
    const { showToast, showConfirmDialog } = vant;

    createApp({
        setup() {
            // 模拟数据
            const shopName = ref("東京ラーメン 新宿店");
            const deadlineTime = ref("14:00");
            const lastOrderConfirmed = ref(false); // 关键：控制是否允许继续订货
            const products = ref([
                { id: 1, name: "特製 醤油タレ", spec: "1L/本", price: 1200, taxRate: 8, stock: 50, image: "https://via.placeholder.com/100" },
                { id: 2, name: "博多細麺", spec: "50玉/箱", price: 4500, taxRate: 8, stock: 0, image: "https://via.placeholder.com/100" },
                { id: 3, name: "掃除用洗剤", spec: "5L/缶", price: 2000, taxRate: 10, stock: 10, image: "https://via.placeholder.com/100" }
            ]);

            const cart = ref([]);
            const showSpecPanel = ref(false);
            const activeItem = ref({});
            const tempQty = ref(1);
            const tempNote = ref("");

            const showConfirmPanel = ref(false);
            const confirmedItems = ref([]);
            const lastOrderItems = [{id: 101, name: "豚骨スープ", qty: 2}, {id: 102, name: "メンマ", qty: 5}];

            // 逻辑处理
            const openSpec = (item) => {
                activeItem.value = item;
                tempQty.value = 1;
                tempNote.value = "";
                showSpecPanel.ref = true;
                showSpecPanel.value = true;
            };

            const calcTaxedPrice = (item) => Math.floor(item.price * (1 + item.taxRate / 100));

            const addToCart = () => {
                cart.value.push({
                    ...activeItem.value,
                    qty: tempQty.value,
                    note: tempNote.value
                });
                showSpecPanel.value = false;
                showToast('カートに追加しました');
            };

            const cartTotalPrice = computed(() => {
                return cart.value.reduce((total, item) => total + (calcTaxedPrice(item) * item.qty), 0);
            });

            const onSubmit = () => {
                showConfirmDialog({
                    title: '注文確定',
                    message: `合計金额 ￥${cartTotalPrice.value} でよろしいですか？`,
                }).then(() => {
                    showToast({ type: 'success', message: '注文完了！工厂正在处理' });
                    cart.value = []; // 清空购物车
                });
            };

            const confirmReceipt = () => {
                if(confirmedItems.value.length < lastOrderItems.length) {
                    showToast('すべての商品を確認してください');
                    return;
                }
                lastOrderConfirmed.value = true;
                showConfirmPanel.value = false;
                showToast('受取確認済み。注文可能です。');
            };

            return {
                shopName, deadlineTime, products, cart, showSpecPanel, activeItem, tempQty, tempNote,
                addToCart, openSpec, calcTaxedPrice, cartTotalPrice, onSubmit,
                lastOrderConfirmed, showConfirmPanel, lastOrderItems, confirmedItems, confirmReceipt
            };
        }
    }).
