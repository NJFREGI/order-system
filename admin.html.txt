<script>
    const { createApp, ref } = Vue;
    const { showToast, showPromptDialog } = vant;

    createApp({
        setup() {
            const deadline = ref("14:00");
            const isOrderingSystemOpen = ref(true);
            const activeTab = ref(0);

            // 模拟订单数据
            const pendingOrders = ref([
                {
                    id: "ORD-001",
                    shopName: "東京ラーメン 新宿店",
                    items: [
                        { name: "特製 醤油タレ", qty: 5, price: 1200 },
                        { name: "掃除用洗剤", qty: 1, price: 2000 }
                    ],
                    note: "急ぎです！"
                }
            ]);

            const markShortage = (order, idx) => {
                order.items[idx].name += " (欠品)";
                order.items[idx].qty = 0;
                showToast('欠品としてマークしました');
                // 这里未来会写同步到数据库的代码
            };

            const printDeliveryNote = (order) => {
                const { jsPDF } = window.jspdf;
                const doc = new jsPDF();
                doc.text("Delivery Note / 納品書", 10, 10);
                doc.text(`Shop: ${order.shopName}`, 10, 20);
                let y = 40;
                order.items.forEach(i => {
                    doc.text(`${i.name} x ${i.qty}`, 10, y);
                    y += 10;
                });
                doc.save(`delivery_${order.id}.pdf`);
            };

            return {
                deadline, isOrderingSystemOpen, activeTab, pendingOrders,
                markShortage, printDeliveryNote
            };
        }
    }).use(vant).mount('#app');
</script>
